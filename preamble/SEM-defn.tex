\let\ORGvarrho=\varrho

\let\ORGvartheta=\vartheta

\let\theta=\ORGvartheta
%
\let\ORGvarphi=\varphi
\let\varphi=\phi
\let\phi=\ORGvarphi

\let\ORGvarleq=\leqslant
\let\leqslant=\leq

\let\ORGvargeq=\geqslant
